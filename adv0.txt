        function openWhatsApp(defaultText) {
            const phoneNumber = "962798807173";
            const text = encodeURIComponent(defaultText || "أريد الاستفسار عن منتج من iHerb");
            window.open(`https://wa.me/${phoneNumber}?text=${text}`, '_blank');
        }
        
        // فتح روابط منتجات آي هيرب حسب التصنيف
        function openCategory(category) {
            const links = {
                'vitamins': 'https://jo.iherb.com/c/vitamins',
                'beauty': 'https://jo.iherb.com/c/beauty',
                'nutrition': 'https://jo.iherb.com/c/sports-nutrition',
                'hair': 'https://jo.iherb.com/c/hair-care'
            };
            window.open(links[category], '_blank');
        }
        
        // إضافة تأثيرات عند التمرير
        document.addEventListener('DOMContentLoaded', function() {
            const features = document.querySelectorAll('.feature');
            features.forEach(feature => {
                feature.addEventListener('click', function() {
                    this.style.transform = 'scale(0.98)';
                    setTimeout(() => {
                        this.style.transform = '';
                    }, 200);
                });
            });
            
            // نظام التقييمات
            const stars = document.querySelectorAll('.star');
            const ratingInput = document.getElementById('rating');
            
            stars.forEach(star => {
                star.addEventListener('click', function() {
                    const value = parseInt(this.getAttribute('data-value'));
                    ratingInput.value = value;
                    
                    stars.forEach(s => {
                        if (parseInt(s.getAttribute('data-value')) <= value) {
                            s.classList.add('active');
                        } else {
                            s.classList.remove('active');
                        }
                    });
                });
                
                star.addEventListener('mouseover', function() {
                    const value = parseInt(this.getAttribute('data-value'));
                    
                    stars.forEach(s => {
                        if (parseInt(s.getAttribute('data-value')) <= value) {
                            s.style.color = '#ffc107';
                        } else {
                            s.style.color = '#ddd';
                        }
                    });
                });
                
                star.addEventListener('mouseout', function() {
                    const currentRating = parseInt(ratingInput.value);
                    
                    stars.forEach(s => {
                        if (parseInt(s.getAttribute('data-value')) <= currentRating) {
                            s.style.color = '#ffc107';
                        } else {
                            s.style.color = '#ddd';
                        }
                    });
                });
            });
            
            // معالجة إرسال التقييم
            const ratingForm = document.getElementById('ratingForm');
            const ratingsContainer = document.getElementById('ratingsContainer');
            
            ratingForm.addEventListener('submit', function(e) {
                e.preventDefault();
                
                const name = document.getElementById('name').value;
                const rating = parseInt(ratingInput.value);
                const comment = document.getElementById('comment').value;
                
                if (rating === 0) {
                    alert('يرجى اختيار تقييم من 5 نجوم');
                    return;
                }
                
                // إنشاء عنصر التقييم الجديد
                const ratingItem = document.createElement('div');
                ratingItem.className = 'rating-item';
                
                let starsHtml = '';
                for (let i = 1; i <= 5; i++) {
                    starsHtml += i <= rating ? '★' : '☆';
                }
                
                ratingItem.innerHTML = `
                    <div class="rating-header">
                        <span class="rating-name">${name}</span>
                        <span class="rating-stars">${starsHtml}</span>
                    </div>
                    ${comment ? `<p class="rating-comment">${comment}</p>` : ''}
                `;
                
                // إضافة التقييم الجديد في الأعلى
                ratingsContainer.insertBefore(ratingItem, ratingsContainer.firstChild);
                
                // إعادة تعيين النموذج
                ratingForm.reset();
                ratingInput.value = 0;
                stars.forEach(s => s.classList.remove('active'));
                
                alert('شكراً لك على تقييمك!');
            });
        });
    </script>
